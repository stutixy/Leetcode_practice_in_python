class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        nums1temp = [nums1[i] for i in range(m)]
        

        it1 = 0
        it2 = 0
        it = 0
        while it1<m and it2<n:
            if nums1temp[it1] < nums2[it2]:
                nums1[it] = nums1temp[it1]
                it1 += 1
            else:
                nums1[it] = nums2[it2]
                it2 += 1
            it += 1
        while it1<m:
            nums1[it] = nums1temp[it1]
            it += 1
            it1 += 1
        while it2<n:
            nums1[it] = nums2[it2]
            it += 1
            it2 += 1
